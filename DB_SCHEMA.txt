CREATE TABLE public.bloqueos_doctor (
  id uuid NOT NULL DEFAULT gen_random_uuid(),

  fecha date NOT NULL,

  hora_fin time without time zone,
  motivo text,

  estado text DEFAULT 'programada'::text,
  created_at timestamp with time zone DEFAULT now(),
  creada_por uuid,
  cancelada_por uuid,
  fecha_cancelacion timestamp with time zone,
  metodo_pago text,
  referencia_pago text,
  CONSTRAINT citas_pkey PRIMARY KEY (id),
  CONSTRAINT citas_doctor_id_fkey FOREIGN KEY (doctor_id) REFERENCES public.doctores(id),
  CONSTRAINT citas_sucursal_id_fkey FOREIGN KEY (sucursal_id) REFERENCES public.sucursales(id),
  CONSTRAINT citas_usuario_id_fkey FOREIGN KEY (usuario_id) REFERENCES public.usuarios(id),
  CONSTRAINT citas_creada_por_fkey FOREIGN KEY (creada_por) REFERENCES public.usuarios(id),
  CONSTRAINT citas_cancelada_por_fkey FOREIGN KEY (cancelada_por) REFERENCES public.usuarios(id)
);
CREATE TABLE public.doctores (

CREATE TABLE public.expedientes_medicos (

  CONSTRAINT horarios_doctor_doctor_id_fkey FOREIGN KEY (doctor_id) REFERENCES public.doctores(id),

  user_id uuid,

  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT notifications_pkey PRIMARY KEY (id),

CREATE TABLE public.recepcionistas (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  usuario_id uuid NOT NULL,

  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT recepcionistas_pkey PRIMARY KEY (id),
  CONSTRAINT fk_usuario FOREIGN KEY (usuario_id) REFERENCES public.usuarios(id),
  CONSTRAINT fk_sucursal FOREIGN KEY (sucursal_id) REFERENCES public.sucursales(id)

CREATE TABLE public.sucursales (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  nombre text NOT NULL,
  direccion text,
  telefono text,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT sucursales_pkey PRIMARY KEY (id)
);
CREATE TABLE public.usuarios (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  nombre_completo text NOT NULL,
  correo text UNIQUE,
  telefono text NOT NULL UNIQUE,
  fecha_nacimiento date,
  rol text NOT NULL CHECK (rol = ANY (ARRAY['admin'::text, 'doctor'::text, 'recepcionista'::text, 'paciente'::text])),
  created_at timestamp with time zone DEFAULT now(),
  tipo_sangre text NOT NULL DEFAULT 'A+'::text CHECK (tipo_sangre = ANY (ARRAY['A+'::text, 'A-'::text, 'B+'::text, 'B-'::text, 'O+'::text, 'O-'::text, 'AB+'::text, 'AB-'::text])),
  registrado_por text NOT NULL DEFAULT '1'::text,
  curp text NOT NULL UNIQUE CHECK (length(curp) <= 18),
  password text NOT NULL,
  CONSTRAINT usuarios_pkey PRIMARY KEY (id)
);
CREATE TABLE public.reembolsos (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  cita_id uuid NOT NULL,
  usuario_id uuid NOT NULL,
  monto numeric NOT NULL,
  estado text NOT NULL DEFAULT 'pendiente'::text CHECK (estado = ANY (ARRAY['pendiente'::text, 'completado'::text, 'no_aplicable'::text])),
  motivo_cancelacion text NOT NULL CHECK (motivo_cancelacion = ANY (ARRAY['paciente'::text, 'doctor'::text, 'recepcionista'::text])),
  fecha_solicitud timestamp with time zone DEFAULT now(),
  fecha_procesamiento timestamp with time zone,
  metodo_pago text,
  referencia_pago text,
  referencia_reembolso text,
  notas text,
  CONSTRAINT reembolsos_pkey PRIMARY KEY (id),
  CONSTRAINT reembolsos_cita_id_fkey FOREIGN KEY (cita_id) REFERENCES public.citas(id),
  CONSTRAINT reembolsos_usuario_id_fkey FOREIGN KEY (usuario_id) REFERENCES public.usuarios(id)
);
